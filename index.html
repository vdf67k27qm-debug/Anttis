import * as THREE from 'three'; import { GLTFLoader } from 'three/examples/jsm/loaders/GLTFLoader.js'; const loader = new GLTFLoader(); const character = new THREE.Group(); // 全パーツをまとめるグループ const bodyGroup = new THREE.Group(); // 股下を基準にするためのグループ // --- 設定：マス目単位の座標（ここを書き換えるだけで調整可能！） --- const GRID = { legLength: 4, // 足の長さ（地面から股下まで） shoulder: { x: 5, y: 4, z: 0 }, // 股下から肩への距離 hip: { x: 2, y: 0, z: 0 } // 股下から股関節への距離 }; // パーツ読み込み関数 function loadPart(name, position, parent) { loader.load(`./${name}.glb`, (gltf) => { const model = gltf.scene; // キャラクター視点の左右(X)を反映 model.position.set(position.x, position.y, position.z); parent.add(model); }); } // --- 組み立て実行 --- // 1. 胴体（股下を基準に配置） // ※ 胴体モデル自体の中心が原点なので、股下から上に y 分浮かす loadPart('Body', { x: 0, y: 0, z: 0 }, bodyGroup); // 2. 腕 loadPart('Arm_R', { x: -GRID.shoulder.x, y: GRID.shoulder.y, z: GRID.shoulder.z }, bodyGroup); loadPart('Arm_L', { x: GRID.shoulder.x, y: GRID.shoulder.y, z: GRID.shoulder.z }, bodyGroup); // 3. 足 loadPart('Leg_R', { x: -GRID.hip.x, y: GRID.hip.y, z: GRID.hip.z }, bodyGroup); loadPart('Leg_L', { x: GRID.hip.x, y: GRID.hip.y, z: GRID.hip.z }, bodyGroup); // 最後に全体を足の長さ分持ち上げて地面に立たせる bodyGroup.position.y = GRID.legLength; character.add(bodyGroup); scene.add(character);
