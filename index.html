import * as THREE from 'three'; import { GLTFLoader } from 'three/examples/jsm/loaders/GLTFLoader'; import * as CANNON from 'cannon-es'; // --- 1. 物理ワールドの設定 --- const world = new CANNON.World(); world.gravity.set(0, -9.82, 0); // 重力 // 衝突グループの設定（自分自身とはぶつからないようにする） const GROUP_PLAYER = 1; const GROUP_OTHERS = 2; // --- 2. キャラクター構成 --- let bodyMesh, armLMesh, armRMesh, legLMesh, legRMesh; let bodyBody, armLBody, armRBody, legLBody, legRBody; const loader = new GLTFLoader(); loader.load('Base.glb', (gltf) => { const scene = gltf.scene; // 各パーツの取り出し（ボクセルマックスでの名前に合わせてください） bodyMesh = scene.getObjectByName('Body'); armLMesh = scene.getObjectByName('Arm_L'); armRMesh = scene.getObjectByName('Arm_R'); legLMesh = scene.getObjectByName('Leg_L'); legRMesh = scene.getObjectByName('Leg_R'); // --- 物理ボディの作成 (Body) --- bodyBody = new CANNON.Body({ mass: 5, shape: new CANNON.Box(new CANNON.Vec3(0.5, 0.8, 0.5)), // サイズはモデルに合わせ調整 position: new CANNON.Vec3(0, 5, 0), angularDamping: 0.9, // 回転しにくくする（起き上がりこぼし効果） collisionFilterGroup: GROUP_PLAYER, collisionFilterMask: GROUP_OTHERS // 自分のパーツ以外とだけぶつかる }); world.addBody(bodyBody); // --- 腕の作成と接続 (Arm_L) --- armLBody = createLimb(armLMesh, new CANNON.Vec3(-0.7, 5.5, 0)); const constraintL = new CANNON.PointToPointConstraint( bodyBody, new CANNON.Vec3(-0.6, 0.5, 0), // 胴体の接続点 armLBody, new CANNON.Vec3(0, 0.4, 0) // 腕の接続点 ); world.addConstraint(constraintL); // 右腕、足も同様に createLimb と Constraint で追加... }); // パーツ作成用関数（共通設定） function createLimb(mesh, pos) { const b = new CANNON.Body({ mass: 1, shape: new CANNON.Box(new CANNON.Vec3(0.2, 0.4, 0.2)), position: pos, collisionFilterGroup: GROUP_PLAYER, collisionFilterMask: GROUP_OTHERS // 自分とはぶつからない }); world.addBody(b); return b; } // --- 3. スマホ・iPad操作（ジョイスティック） --- let moveVec = { x: 0, z: 0 }; window.addEventListener('touchmove', (e) => { // 指の動きをベクトルに変換（簡易版） const touch = e.touches[0]; moveVec.x = (touch.clientX / window.innerWidth - 0.5) * 2; moveVec.z = (touch.clientY / window.innerHeight - 0.5) * 2; }, { passive: false }); // --- 4. 更新ループ --- function animate() { requestAnimationFrame(animate); world.fixedStep(); if (bodyBody) { // 移動：倒した方向に力を加える bodyBody.velocity.x = moveVec.x * 5; bodyBody.velocity.z = moveVec.z * 5; // 起き上がりこぼしの魔法：常に上を向かせる力を加える const uprightPower = 10; bodyBody.applyTorque(new CANNON.Vec3( -bodyBody.quaternion.x * uprightPower, 0, -bodyBody.quaternion.z * uprightPower )); // メッシュの位置を物理ボディに合わせる bodyMesh.position.copy(bodyBody.position); bodyMesh.quaternion.copy(bodyBody.quaternion); // ...他のパーツも同様に同期 } } animate();
